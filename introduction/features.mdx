    ```javascript
    ws.on('message', (data) => {
        const event = JSON.parse(data);
        if (event.type === 'price_update') {
            updatePortfolioValue(event.data);
        }
    });
    ```
  </Tab>
  <Tab title="SDK">
    ```python
    # Python SDK
    from guru_network import GuruClient

    client = GuruClient(api_key='your-api-key')

    # Create automated trading strategy
    strategy = client.strategies.create({
        'name': 'Arbitrage Bot',
        'type': 'arbitrage',
        'exchanges': ['uniswap', 'sushiswap'],
        'min_profit': 0.5  # 0.5% minimum profit
    })

    # Start strategy execution
    strategy.start()
    ```
